P2607\sf\color{red}{P2607}P2607

nnn 个节点 nnn 条边是基环树森林。

如果树的最大独立集问题那就很简单了，但是不是，所以不会。

看 @leaving@\text leaving@leaving 写的，学习基环树。

对于这个环，取一个环上的边 (lt,rt)(lt,rt)(lt,rt) 。

那么有 333 种情况 ，一是 ltltlt 选 rtrtrt 不选，二是 ltltlt 不选 rtrtrt 选，三是 lt,rtlt , rtlt,rt 都不选 。

就舍弃 (lt,rt)(lt,rt)(lt,rt) 以 ltltlt 和 rtrtrt 为根分别跑一遍树形 DPDPDP 。

用并查集找环比较方便 。

#include <iostream>

using namespace std;

#define int long long

struct edge
{
    int to,next;
}e[2000005];
int cnt,head[1000005];

int n;
int fa[1000005],dp[1000005][2],w[1000005];

void add(int x,int y)
{
    cnt++;
    e[cnt].to=y;
    e[cnt].next=head[x];
    head[x]=cnt;
}

int findx(int x)
{
    if(x==fa[x])return x;
    return fa[x]=findx(fa[x]);
}

int L[1000005],R[1000005],tot;

int ans;

void dfs(int x,int fa)
{
    dp[x][1]=w[x];
    dp[x][0]=0;
    for(int i=head[x];i;i=e[i].next)
    {
        int v=e[i].to;
        if(v==fa)continue;
        dfs(v,x);
        dp[x][1]+=dp[v][0];
        dp[x][0]+=max(dp[v][0],dp[v][1]);
    }
}

signed main()
{
    int i,j,k;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        fa[i]=i;
    }
    for(i=1;i<=n;i++)
    {
        int x;
        cin>>w[i]>>x;
        int fi=findx(i),fx=findx(x);
        if(fi!=fx)fa[fi]=fx,add(i,x),add(x,i);
        else
        {
            tot++;
            L[tot]=i;
            R[tot]=x;
        }
    }
    for(i=1;i<=tot;i++)
    {
        dfs(L[i],-1);
        int t1=dp[L[i]][0];
        dfs(R[i],-1);
        int t2=dp[R[i]][0];
        ans+=max(t1,t2);
    }
    cout<<ans;
    return 0;
}

#include <bits/stdc++.h>

using namespace std;

int n,m;
char s[3005][3005];

int main()
{
    int i,j,k;
    cin>>n>>m;
    for(i=1;i<=n;i++) 
    {
        cin>>s[i]+1;
    }
    int ans=0;
    for (i=1;i<=n+m-1;i++) 
    {
        int x,y;
        x=min(i,n);
        y=max(1,i-n+1);
        int A=0,B=0,C=0;
        while(x&&y<=m) 
        {
            int g=C;
            C=max(C,max(A, B));
            if(s[x][y-1]=='R'&&s[x][y]=='G'&&s[x][y+1]=='W')A=max(A,g)+1;
            if(s[x-1][y]=='R'&&s[x][y]=='G'&&s[x+1][y]=='W')B=max(B,g)+1;
            x--;
            y++;
        }
        ans+=max(A,max(B,C));
    }
    cout<<ans;
    return 0;
}
